JDBC -> Java Database Connectivity
=========================
To store and retrieve data, you can use an RDBMS. MySQL has been the most preferred RDBMS for developing Web applications on an open-source platform
Dr. E. F. Codd proposed the relational model for database systems in 1970.
 It is the basis for the relational database management system (RDBMS).
Steps to Connect with Backend
Register the driver class
Creating connection
Creating statement
Executing queries
Closing connection
SQL COMMANDS:
 	(DDL(Data Definition Language) - Create, Alter, Drop
	DML(Data Manipulation Language)- Insert, Update, Delete
	TCL(Transaction and control Language) - commit, Rollback)
The forName() method of Class class is used to register the driver class
The getConnection() method of DriverManager class is used to 
      establish connection with the database.
The createStatement() method of Connection interface is used to 
create statement

The executeQuery() method of Statement interface is used to execute queries to database.

 ResultSet that can be used to get all the records of a table
Example to close connection

con.close();  

EX:
package example01;

import java.sql.*;
import java.util.*;
public class Example01 {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        // TODO code application logic here
    int eid1,sal1;
        String ename1,sql1;
            int ch;
 Scanner obj=new Scanner(System.in);
             
 try
 {
     Class.forName("com.mysql.jdbc.Driver");
 Connection con=DriverManager.getConnection("jdbc:mysql://localhost/emp","root","");  
 
   while(true)
   {
     System.out.println("1.INSERT");
     System.out.println("2.DELETE");
     System.out.println("3.UPDATE");
     System.out.println("4.SELECT");
     System.out.println("5.EXIT");
     System.out.println("Enter Your Choic [1-5]");
     ch=obj.nextInt();
     switch(ch)
     {
         case 1:
             System.out.println("1.INSERT");
             System.out.println("Enter eid,ename,sal");
    eid1=obj.nextInt();
    ename1=obj.next();
    sal1=obj.nextInt();
              sql1 = "INSERT INTO emp_details(eid,ename,sal) VALUES (?, ?, ?)";
               PreparedStatement statement=con.prepareStatement(sql1);
               statement.setInt(1,eid1);
               statement.setString(2,ename1);
               statement.setInt(3,sal1);

               int rowsInserted = statement.executeUpdate();
               if (rowsInserted > 0)
               {
                    System.out.println("A new employee was inserted successfully!\n");
               }
             break;
         case 2:
             System.out.println("2.DELETE");
             System.out.println("Enter eid:");
             eid1=obj.nextInt();
              String sql2 = "DELETE FROM emp_details WHERE eid=?";
               PreparedStatement statement1 = con.prepareStatement(sql2);
               statement1.setInt(1,eid1);
               statement1.executeUpdate();
               System.out.println("A Employee was deleted successfully!\n");
               break;
          case 3:
             System.out.println("3.UPDATE");
             System.out.println("Enter eid,sal:");
    eid1=obj.nextInt();
    sal1=obj.nextInt();
    String sql3 = "Update emp_details set sal=? where eid=?";
               PreparedStatement pstmt = con.prepareStatement(sql3);
               pstmt.setInt(1,sal1);
               pstmt.setInt(2,eid1);
               pstmt.executeUpdate();
               System.out.println("\nRecord updated successfully!!\n");
               break;
          case 4:
             System.out.println("4.SELECT");
              String sql5 = "SELECT * FROM emp_details";
               Statement stmt1 = con.createStatement();
               ResultSet result1 = stmt1.executeQuery(sql5);

               while (result1.next())
               {
                    System.out.println(result1.getInt(1)+" "+
                    result1.getString(2)+" "+
                    result1.getInt(3));
               }
             break;
          case 5:
System.out.println("Thank You");
System.exit(0);
break;
          default:
System.out.println("Invalid Selection");
break;
     }
   }
 }
 catch(Exception ex)
 {
 }
 
 }
   
   
}
